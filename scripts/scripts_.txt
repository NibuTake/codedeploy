#!/bin/bash
sudo yum -y groupinstall 'Development tools'
sudo yum -y install zlib-devel openssl-devel

prefix='/home/ec2-user/'
mkdir ${prefix}python3.6
cd ${prefix}python3.6
wget https://www.python.org/ftp/python/3.6.1/Python-3.6.1.tgz
tar zxvf Python-3.6.1.tgz
cd ${prefix}/python3.6/Python-3.6.1
./configure
make
sudo make install


sudo /usr/local/bin/pip3.6 install virtualenv

/usr/local/bin/virtualenv -p /home/ec2-user/python3.6 encryptfile

cd encryptfile/
source bin/activate

pip install pandas
